module Up_Dn_Counter_tb ()

/////////////////////////////////////////////////////
//////////////////clk_generator//////////////////////
/////////////////////////////////////////////////////

parameter clk_period= 20; 
reg clk_tb=0; 
always #(clk_period/2) clk_tb = ~clk_tb;

/////////////////////////////////////////////////////
///////////////Decleration & Instances///////////////
/////////////////////////////////////////////////////

wire [3:0]	IN_tb; 
wire		Load_tb;
wire		Up_tb;
wire		Down_tb;
wire		CLK_tb;
reg			High_tb;
reg	[3:0]	Counter_tb; 
reg			Low_tb;

 Up_Dn_Counter DUT (
		.IN(IN_tb),
		.Load(Load_tb),
		.Up(Up_tb),
		.Down(Down_tb),
		.CLK(CLK_tb),
		.rst(rst_tb),
		.High(High_tb),
		.Counter(Counter_tb),
		.Low(Low_tb)
		);


/////////////////////////////////////////////////////
///////////////////Initial Block/////////////////////
/////////////////////////////////////////////////////

initial begin 
	$dumpfile("Up_Dn_Counter.vcd"); 
	$dumpvars; 
end 

/////////////////////////////////////////////////////
//////////////////////Tasks//////////////////////////
/////////////////////////////////////////////////////

task reset;
 begin
 rst_tb=1;
 #(clk_period)
 rst_tb=0;
 #(clk_period)
 rst_tb=1;
 end
endtask
